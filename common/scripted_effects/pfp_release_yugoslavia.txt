#Scripted effect to release Yugoslavia (P50)
pfp_release_yugoslavia = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_enabled }

			#Add contestion to formable puppet
			45 = { add_contested_owner  = P50 } #Batschka
			105 = { add_contested_owner  = P50 } #Crna Gora
			107 = { add_contested_owner  = P50 } #Šumadija
			108 = { add_contested_owner  = P50 } #Pomoravlje
			764 = { add_contested_owner  = P50 } #West Banat
			952 = { add_contested_owner  = P50 } #Syrmia
			963 = { add_contested_owner  = P50 } #Raška

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = P50 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = P50
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = P50
				}
			}

			#Set custom focus tree to combat resistance
			P50 = { load_focus_tree = pfp_formable_puppet_focus }
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			45 = { add_core_of = P50 } #Batschka
			105 = { add_core_of = P50 } #Crna Gora
			107 = { add_core_of = P50 } #Šumadija
			108 = { add_core_of = P50 } #Pomoravlje
			764 = { add_core_of = P50 } #West Banat
			952 = { add_core_of = P50 } #Syrmia
			963 = { add_core_of = P50 } #Raška

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = P50 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = P50
					transfer_state_to = P50
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = P50
					set_state_controller_to = P50
				}
			}

			P50 = {
				remove_ideas = {
					pfp_puppet_resistance_idea_1
				}
			}
		}
	}
}
