#Scripted effect to release Lombardy-Venetia (P92)
pfp_release_lombardy-venetia = {
    hidden_effect = {
        #Add cores to Lombardy-Venetia
        159 = { add_core_of = P92 } #Lombardia
        160 = { add_core_of = P92 } #Veneto
        912 = { add_core_of = P92 } #Friuli
        
        #Set formable puppet as owner and controller of all its cored states
        every_state = {
            limit = { is_core_of = P92 }
            add_contested_owner = P92
            transfer_state_to = P92
        }

        #Transfer all technology to formable puppet
        P92 = { pfp_transfer_technology = yes }

        #Set global flag to prevent future decision use
        set_global_flag = pfp_lombardy-venetia_flag

        #Send event to all nations about the formable puppet
        set_temp_variable = { temp_country_tag = P92 }
    }
}
