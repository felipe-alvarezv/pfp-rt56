#Scripted effect to release Finland (P12)
pfp_release_finland = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_enabled }

			#Add contestion to formable puppet
			111 = { add_contested_owner = P12 } #Uusimaa
			148 = { add_contested_owner = P12 } #Lappi
			149 = { add_contested_owner = P12 } #Pohjanmaa
			150 = { add_contested_owner = P12 } #Savo
			1036 = { add_contested_owner = P12 } #Turku
			1037 = { add_contested_owner = P12 } #Hame
			1038 = { add_contested_owner = P12 } #Kymi
			1039 = { add_contested_owner = P12 } #Oulu
			1040 = { add_contested_owner = P12 } #Mikkeli

			#Add claims to formable puppet
			111 = { add_claim_by = P12 } #Uusimaa
			148 = { add_claim_by = P12 } #Lappi
			149 = { add_claim_by = P12 } #Pohjanmaa
			150 = { add_claim_by = P12 } #Savo
			1036 = { add_claim_by = P12 } #Turku
			1037 = { add_claim_by = P12 } #Hame
			1038 = { add_claim_by = P12 } #Kymi
			1039 = { add_claim_by = P12 } #Oulu
			1040 = { add_claim_by = P12 } #Mikkeli

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = P12 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = P12
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = P12
				}
			}

			#Set custom focus tree to combat resistance
			P12 = { load_focus_tree = pfp_formable_puppet_focus }
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			111 = { add_core_of = P12 } #Uusimaa
			148 = { add_core_of = P12 } #Lappi
			149 = { add_core_of = P12 } #Pohjanmaa
			150 = { add_core_of = P12 } #Savo
			1036 = { add_core_of = P12 } #Turku
			1037 = { add_core_of = P12 } #Hame
			1038 = { add_core_of = P12 } #Kymi
			1039 = { add_core_of = P12 } #Oulu
			1040 = { add_core_of = P12 } #Mikkeli

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = P12 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = P12
					transfer_state_to = P12
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = P12
					set_state_controller_to = P12
				}
			}

			P12 = {
				remove_ideas = {
					pfp_puppet_resistance_idea_1
				}
			}
		}
	}
}
