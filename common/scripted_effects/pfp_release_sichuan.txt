#Scripted effect to release Sichuan (Q19)
pfp_release_sichuan = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_enabled }

			#Add contestion to formable puppet
			605 = { add_contested_owner  = Q19 } #Sichuan
			751 = { add_contested_owner  = Q19 } #Liangshan
			752 = { add_contested_owner  = Q19 } #Ganzi
			753 = { add_contested_owner  = Q19 } #Gannan
			1116 = { add_contested_owner  = Q19 } #Chongqing
			1120 = { add_contested_owner  = Q19 } #Chengdu

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = Q19 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = Q19
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = Q19
				}
			}

			#Set custom focus tree to combat resistance
			Q19 = { load_focus_tree = pfp_formable_puppet_focus }
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			605 = { add_core_of = Q19 } #Sichuan
			751 = { add_core_of = Q19 } #Liangshan
			752 = { add_core_of = Q19 } #Ganzi
			753 = { add_core_of = Q19 } #Gannan
			1116 = { add_core_of = Q19 } #Chongqing
			1120 = { add_core_of = Q19 } #Chengdu

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = Q19 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = Q19
					transfer_state_to = Q19
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = Q19
					set_state_controller_to = Q19
				}
			}

			Q19 = {
				remove_ideas = {
					pfp_puppet_resistance_idea_1
				}
			}
		}
	}
}
