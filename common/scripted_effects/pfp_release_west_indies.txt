#Scripted effect to release West Indies (P06)
pfp_release_west_indies = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_enabled }

			#Add contestion to formable puppet
			315 = { add_contested_owner = P06 } #Santiago de Cuba
			318 = { add_contested_owner = P06 } #Haiti
			319 = { add_contested_owner = P06 } #Dominican Republic
			689 = { add_contested_owner = P06 } #Jamaica
			994 = { add_contested_owner = P06 } #La Habana

			#Add claims to formable puppet
			315 = { add_claim_by = P06 } #Santiago de Cuba
			318 = { add_claim_by = P06 } #Haiti
			319 = { add_claim_by = P06 } #Dominican Republic
			689 = { add_claim_by = P06 } #Jamaica
			994 = { add_claim_by = P06 } #La Habana

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = P06 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = P06
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = P06
				}
			}

			#Set custom focus tree to combat resistance
			P06 = { load_focus_tree = pfp_formable_puppet_focus }
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			315 = { add_core_of = P06 } #Santiago de Cuba
			318 = { add_core_of = P06 } #Haiti
			319 = { add_core_of = P06 } #Dominican Republic
			689 = { add_core_of = P06 } #Jamaica
			994 = { add_core_of = P06 } #La Habana

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = P06 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = P06
					transfer_state_to = P06
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = P06
					set_state_controller_to = P06
				}
			}

			P06 = {
				remove_ideas = {
					pfp_puppet_resistance_idea_1
				}
			}
		}
	}
}
