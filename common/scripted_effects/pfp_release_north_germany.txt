#Scripted effect to release North Germany (P19)
pfp_release_north_germany = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_enabled }

			#Add contestion to formable puppet
			989 = { add_contested_owner  = P19 } #Ostrhineland
			51 = { add_contested_owner  = P19 } #Rhineland
			55 = { add_contested_owner  = P19 } #Hessen
			56 = { add_contested_owner  = P19 } #Weser-Ems
			57 = { add_contested_owner  = P19 } #Westfalen
			58 = { add_contested_owner  = P19 } #Holstein
			59 = { add_contested_owner  = P19 } #Hannover
			60 = { add_contested_owner  = P19 } #Thuringen
			61 = { add_contested_owner  = P19 } #Mecklenburg
			62 = { add_contested_owner  = P19 } #Vorpommern
			63 = { add_contested_owner  = P19 } #Hinterpommern
			64 = { add_contested_owner  = P19 } #Brandenburg
			65 = { add_contested_owner  = P19 } #Sachsen
			66 = { add_contested_owner  = P19 } #Niederschlesien
			67 = { add_contested_owner  = P19 } #Oberschlesien
			68 = { add_contested_owner  = P19 } #Neumark

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = P19 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = P19
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = P19
				}
			}

			#Set custom focus tree to combat resistance
			P19 = { load_focus_tree = pfp_formable_puppet_focus }
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			989 = { add_core_of = P19 } #Ostrhineland
			51 = { add_core_of = P19 } #Rhineland
			55 = { add_core_of = P19 } #Hessen
			56 = { add_core_of = P19 } #Weser-Ems
			57 = { add_core_of = P19 } #Westfalen
			58 = { add_core_of = P19 } #Holstein
			59 = { add_core_of = P19 } #Hannover
			60 = { add_core_of = P19 } #Thuringen
			61 = { add_core_of = P19 } #Mecklenburg
			62 = { add_core_of = P19 } #Vorpommern
			63 = { add_core_of = P19 } #Hinterpommern
			64 = { add_core_of = P19 } #Brandenburg
			65 = { add_core_of = P19 } #Sachsen
			66 = { add_core_of = P19 } #Niederschlesien
			67 = { add_core_of = P19 } #Oberschlesien
			68 = { add_core_of = P19 } #Neumark

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = P19 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = P19
					transfer_state_to = P19
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = P19
					set_state_controller_to = P19
				}
			}

			P19 = {
				remove_ideas = {
					pfp_puppet_resistance_idea_1
				}
			}
		}
	}
}
