#Scripted effect to release Mexico (P58)
pfp_release_mexico = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_enabled }

			#Add contestion to formable puppet
			277 = { add_contested_owner = P58 } #Mexico City
			474 = { add_contested_owner = P58 } #Yucatan
			475 = { add_contested_owner = P58 } #Chiapas
			476 = { add_contested_owner = P58 } #Oaxaca
			477 = { add_contested_owner = P58 } #Veracruz
			478 = { add_contested_owner = P58 } #Jalisco
			479 = { add_contested_owner = P58 } #Tamauolipas
			480 = { add_contested_owner = P58 } #Coahuila
			481 = { add_contested_owner = P58 } #Sinaloa
			482 = { add_contested_owner = P58 } #Chihuahua
			483 = { add_contested_owner = P58 } #Sonora
			484 = { add_contested_owner = P58 } #Baja California
			485 = { add_contested_owner = P58 } #Guerrero
			929 = { add_contested_owner = P58 } #Durango
			930 = { add_contested_owner = P58 } #Zacatecas
			931 = { add_contested_owner = P58 } #San Luis Potosi
			932 = { add_contested_owner = P58 } #Puebla

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = P58 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = P58
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = P58
				}
			}

			#Set custom focus tree to combat resistance
			P58 = { load_focus_tree = pfp_formable_puppet_focus }
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			277 = { add_core_of = P58 } #Mexico City
			474 = { add_core_of = P58 } #Yucatan
			475 = { add_core_of = P58 } #Chiapas
			476 = { add_core_of = P58 } #Oaxaca
			477 = { add_core_of = P58 } #Veracruz
			478 = { add_core_of = P58 } #Jalisco
			479 = { add_core_of = P58 } #Tamauolipas
			480 = { add_core_of = P58 } #Coahuila
			481 = { add_core_of = P58 } #Sinaloa
			482 = { add_core_of = P58 } #Chihuahua
			483 = { add_core_of = P58 } #Sonora
			484 = { add_core_of = P58 } #Baja California
			485 = { add_core_of = P58 } #Guerrero
			929 = { add_core_of = P58 } #Durango
			930 = { add_core_of = P58 } #Zacatecas
			931 = { add_core_of = P58 } #San Luis Potosi
			932 = { add_core_of = P58 } #Puebla

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = P58 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = P58
					transfer_state_to = P58
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = P58
					set_state_controller_to = P58
				}
			}

			P58 = {
				remove_ideas = {
					pfp_puppet_resistance_idea_1
				}
			}
		}
	}
}
