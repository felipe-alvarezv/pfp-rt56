#Scripted effect to release Arabia (P03)
pfp_release_arabia = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_enabled }

			#Add contestion to formable puppet
			292 = { add_contested_owner = P03 } #Nejd
			678 = { add_contested_owner = P03 } #Rub Al Khali
			679 = { add_contested_owner = P03 } #Madinah
			854 = { add_contested_owner = P03 } #Jawf
			855 = { add_contested_owner = P03 } #Tabuk
			856 = { add_contested_owner = P03 } #Asir-Makkah
			857 = { add_contested_owner = P03 } #Al-Qassim
			859 = { add_contested_owner = P03 } #Dammam

			#Add claims to formable puppet
			292 = { add_claim_by = P03 } #Nejd
			678 = { add_claim_by = P03 } #Rub Al Khali
			679 = { add_claim_by = P03 } #Madinah
			854 = { add_claim_by = P03 } #Jawf
			855 = { add_claim_by = P03 } #Tabuk
			856 = { add_claim_by = P03 } #Asir-Makkah
			857 = { add_claim_by = P03 } #Al-Qassim
			859 = { add_claim_by = P03 } #Dammam

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = P03 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = P03
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = P03
				}
			}

			#Set custom focus tree to combat resistance
			P03 = { load_focus_tree = pfp_formable_puppet_focus }
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			292 = { add_core_of = P03 } #Nejd
			678 = { add_core_of = P03 } #Rub Al Khali
			679 = { add_core_of = P03 } #Madinah
			854 = { add_core_of = P03 } #Jawf
			855 = { add_core_of = P03 } #Tabuk
			856 = { add_core_of = P03 } #Asir-Makkah
			857 = { add_core_of = P03 } #Al-Qassim
			859 = { add_core_of = P03 } #Dammam

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = P03 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = P03
					transfer_state_to = P03
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = P03
					set_state_controller_to = P03
				}
			}

			P03 = {
				remove_ideas = {
					pfp_puppet_resistance_idea_1
				}
			}
		}
	}
}
