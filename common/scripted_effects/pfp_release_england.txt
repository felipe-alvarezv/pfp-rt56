#Scripted effect to release England (P80)
pfp_release_england = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_enabled }

			#Add contestion to formable puppet
			123 = { add_contested_owner = P80 } #South-West England
			125 = { add_contested_owner = P80 } #East Anglia
			126 = { add_contested_owner = P80 } #Greater London Area
			127 = { add_contested_owner = P80 } #Sussex
			128 = { add_contested_owner = P80 } #West Midlands
			129 = { add_contested_owner = P80 } #East Midlands
			130 = { add_contested_owner = P80 } #Yorkshire
			131 = { add_contested_owner = P80 } #Northumberland
			132 = { add_contested_owner = P80 } #Lancashire
			338 = { add_contested_owner = P80 } #Gloucestershire
			1041 = { add_contested_owner = P80 } #Cumbria

			#Add claims to formable puppet
			123 = { add_claim_by = P80 } #South-West England
			125 = { add_claim_by = P80 } #East Anglia
			126 = { add_claim_by = P80 } #Greater London Area
			127 = { add_claim_by = P80 } #Sussex
			128 = { add_claim_by = P80 } #West Midlands
			129 = { add_claim_by = P80 } #East Midlands
			130 = { add_claim_by = P80 } #Yorkshire
			131 = { add_claim_by = P80 } #Northumberland
			132 = { add_claim_by = P80 } #Lancashire
			338 = { add_claim_by = P80 } #Gloucestershire
			1041 = { add_claim_by = P80 } #Cumbria

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = P80 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = P80
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = P80
				}
			}

			#Set custom focus tree to combat resistance
			P80 = { load_focus_tree = pfp_formable_puppet_focus }
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			123 = { add_core_of = P80 } #South-West England
			125 = { add_core_of = P80 } #East Anglia
			126 = { add_core_of = P80 } #Greater London Area
			127 = { add_core_of = P80 } #Sussex
			128 = { add_core_of = P80 } #West Midlands
			129 = { add_core_of = P80 } #East Midlands
			130 = { add_core_of = P80 } #Yorkshire
			131 = { add_core_of = P80 } #Northumberland
			132 = { add_core_of = P80 } #Lancashire
			338 = { add_core_of = P80 } #Gloucestershire
			1041 = { add_core_of = P80 } #Cumbria

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = P80 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = P80
					transfer_state_to = P80
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = P80
					set_state_controller_to = P80
				}
			}

			P80 = {
				remove_ideas = {
					pfp_puppet_resistance_idea_1
				}
			}
		}
	}
}
