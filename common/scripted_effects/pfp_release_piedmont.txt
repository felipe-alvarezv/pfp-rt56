#Scripted effect to release Piedmont (P93)
pfp_release_piedmont = {
    hidden_effect = {
        #Add cores to Piedmont
        158 = { add_core_of = P93 } #Piedmonte
        
        #Set formable puppet as owner and controller of all its cored states
        every_state = {
            limit = { is_core_of = P93 }
            add_contested_owner = P93
            transfer_state_to = P93
        }

        #Transfer all technology to formable puppet
        P93 = { pfp_transfer_technology = yes }

        #Set global flag to prevent future decision use
        set_global_flag = pfp_piedmont_flag

        #Send event to all nations about the formable puppet
        set_temp_variable = { temp_country_tag = P93 }
    }
}
