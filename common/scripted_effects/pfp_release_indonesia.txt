#Scripted effect to release Indonesia (P17)
pfp_release_indonesia = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_enabled }

			#Add contestion to formable puppet
			334 = { add_contested_owner = P17 } #East Kalimantan
			335 = { add_contested_owner = P17 } #West Java
			667 = { add_contested_owner = P17 } #Lesser Sunda Islands
			668 = { add_contested_owner = P17 } #Moluccan Islands
			672 = { add_contested_owner = P17 } #South Sumatra
			673 = { add_contested_owner = P17 } #South Sulawesi
			738 = { add_contested_owner = P17 } #Aru Islands
			974 = { add_contested_owner = P17 } #North Sumatra
			976 = { add_contested_owner = P17 } #Central Java
			446 = { add_contested_owner = P17 } #East Java
			975 = { add_contested_owner = P17 } #North Sulawesi
			977 = { add_contested_owner = P17 } #Inner Borneo
			978 = { add_contested_owner = P17 } #West Kalimantan

			#Add claims to formable puppet
			334 = { add_claim_by = P17 } #East Kalimantan
			335 = { add_claim_by = P17 } #West Java
			667 = { add_claim_by = P17 } #Lesser Sunda Islands
			668 = { add_claim_by = P17 } #Moluccan Islands
			672 = { add_claim_by = P17 } #South Sumatra
			673 = { add_claim_by = P17 } #South Sulawesi
			738 = { add_claim_by = P17 } #Aru Islands
			974 = { add_claim_by = P17 } #North Sumatra
			976 = { add_claim_by = P17 } #Central Java
			446 = { add_claim_by = P17 } #East Java
			975 = { add_claim_by = P17 } #North Sulawesi
			977 = { add_claim_by = P17 } #Inner Borneo
			978 = { add_claim_by = P17 } #West Kalimantan

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = P17 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = P17
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = P17
				}
			}

			#Set custom focus tree to combat resistance
			P17 = { load_focus_tree = pfp_formable_puppet_focus }
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			334 = { add_core_of = P17 } #East Kalimantan
			335 = { add_core_of = P17 } #West Java
			667 = { add_core_of = P17 } #Lesser Sunda Islands
			668 = { add_core_of = P17 } #Moluccan Islands
			672 = { add_core_of = P17 } #South Sumatra
			673 = { add_core_of = P17 } #South Sulawesi
			738 = { add_core_of = P17 } #Aru Islands
			974 = { add_core_of = P17 } #North Sumatra
			976 = { add_core_of = P17 } #Central Java
			446 = { add_core_of = P17 } #East Java
			975 = { add_core_of = P17 } #North Sulawesi
			977 = { add_core_of = P17 } #Inner Borneo
			978 = { add_core_of = P17 } #West Kalimantan

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = P17 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = P17
					transfer_state_to = P17
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = P17
					set_state_controller_to = P17
				}
			}

			P17 = {
				remove_ideas = {
					pfp_puppet_resistance_idea_1
				}
			}
		}
	}
}
