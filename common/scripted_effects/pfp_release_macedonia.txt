#Scripted effect to release Macedonia (P98)
pfp_release_macedonia = {
    hidden_effect = {
        #Add cores to Macedonia
        106 = { add_core_of = P98 } #Makedonija
        950 = { add_core_of = P98 } #Tetovo
        
        #Set formable puppet as owner and controller of all its cored states
        every_state = {
            limit = { is_core_of = P98 }
            add_contested_owner = P98
            transfer_state_to = P98
        }

        #Transfer all technology to formable puppet
        P98 = { pfp_transfer_technology = yes }

        #Set global flag to prevent future decision use
        set_global_flag = pfp_macedonia_flag

        #Send event to all nations about the formable puppet
        set_temp_variable = { temp_country_tag = P98 }
    }
}
