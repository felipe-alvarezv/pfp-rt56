#Scripted effect to release Afghanistan (P76)
pfp_release_afghanistan = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_enabled }

			#Add contestion to formable puppet
			267 = { add_contested_owner  = P76 } #Kabulistan
			415 = { add_contested_owner  = P76 } #Herat
			1008 = { add_contested_owner  = P76 } #Faryab
			1009 = { add_contested_owner  = P76 } #Bamian
			1010 = { add_contested_owner  = P76 } #Kandahar
			1011 = { add_contested_owner  = P76 } #Badakhshan

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = P76 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = P76
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = P76
				}
			}

			#Set custom focus tree to combat resistance
			P76 = { load_focus_tree = pfp_formable_puppet_focus }
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			267 = { add_core_of = P76 } #Kabulistan
			415 = { add_core_of = P76 } #Herat
			1008 = { add_core_of = P76 } #Faryab
			1009 = { add_core_of = P76 } #Bamian
			1010 = { add_core_of = P76 } #Kandahar
			1011 = { add_core_of = P76 } #Badakhshan

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = P76 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = P76
					transfer_state_to = P76
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = P76
					set_state_controller_to = P76
				}
			}

			P76 = {
				remove_ideas = {
					pfp_puppet_resistance_idea_1
				}
			}
		}
	}
}
