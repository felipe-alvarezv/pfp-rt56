#Scripted effect to release Qinghai (Q21)
pfp_release_qinghai = {
    hidden_effect = {
        #Add cores to Qinghai
        604 = { add_core_of = Q21 } #Qinghai
        754 = { add_core_of = Q21 } #Golog
        755 = { add_core_of = Q21 } #Haixi
        1075 = { add_core_of = Q21 } #Gyarong
        
        #Set formable puppet as owner and controller of all its cored states
        every_state = {
            limit = { is_core_of = Q21 }
            add_contested_owner = Q21
            transfer_state_to = Q21
        }

        #Transfer all technology to formable puppet
        Q21 = { pfp_transfer_technology = yes }

        #Set global flag to prevent future decision use
        set_global_flag = pfp_qinghai_flag

        #Send event to all nations about the formable puppet
        set_temp_variable = { temp_country_tag = Q21 }
    }
}
