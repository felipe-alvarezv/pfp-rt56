pfp_release_shaan-gan = {
    hidden_effect = {
        #Add cores to Shaan-Gan
        283 = { add_core_of = Q20 } #Gansu
        622 = { add_core_of = Q20 } #Shaanbei
        744 = { add_core_of = Q20 } #Xian
        756 = { add_core_of = Q20 } #Jiuquan
        1111 = { add_core_of = Q20 } #Yan'an
        1123 = { add_core_of = Q20 } #Wuwei
        1124 = { add_core_of = Q20 } #Guyuan
        1125 = { add_core_of = Q20 } #Yulin
        
        #Set formable puppet as owner and controller of all its cored states
        every_state = {
            limit = { is_core_of = Q20 }
            add_contested_owner = Q20
            transfer_state_to = Q20
        }

        #Transfer all technology to formable puppet
        Q20 = { pfp_transfer_technology = yes }

        #Set global flag to prevent future decision use
        set_global_flag = pfp_shaan-gan_flag

        #Send event to all nations about the formable puppet
        set_temp_variable = { temp_country_tag = Q20 }
    }
}
