#Scripted effect to release Central America (P08)
pfp_release_central_america = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_enabled }

			#Add contestion to formable puppet
			304 = { add_contested_owner  = P08 } #Panama
			312 = { add_contested_owner  = P08 } #Honduras
			313 = { add_contested_owner  = P08 } #Guatemala
			314 = { add_contested_owner  = P08 } #El Salvador
			316 = { add_contested_owner  = P08 } #Costa Rica
			317 = { add_contested_owner  = P08 } #Nicaragua
			937 = { add_contested_owner  = P08 } #Western Panama

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = P08 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = P08
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = P08
				}
			}

			#Set custom focus tree to combat resistance
			P08 = { load_focus_tree = pfp_formable_puppet_focus }
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			304 = { add_core_of = P08 } #Panama
			312 = { add_core_of = P08 } #Honduras
			313 = { add_core_of = P08 } #Guatemala
			314 = { add_core_of = P08 } #El Salvador
			316 = { add_core_of = P08 } #Costa Rica
			317 = { add_core_of = P08 } #Nicaragua
			937 = { add_core_of = P08 } #Western Panama

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = P08 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = P08
					transfer_state_to = P08
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = P08
					set_state_controller_to = P08
				}
			}

			P08 = {
				remove_ideas = {
					pfp_puppet_resistance_idea_1
				}
			}
		}
	}
}
