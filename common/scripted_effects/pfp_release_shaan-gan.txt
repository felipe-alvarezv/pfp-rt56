#Scripted effect to release Shaan-Gan (Q20)
pfp_release_shaan-gan = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_enabled }

			#Add contestion to formable puppet
			283 = { add_contested_owner  = Q20 } #Gansu
			622 = { add_contested_owner  = Q20 } #Shaanbei
			744 = { add_contested_owner  = Q20 } #Xian
			756 = { add_contested_owner  = Q20 } #Jiuquan
			1111 = { add_contested_owner  = Q20 } #Yan'an
			1123 = { add_contested_owner  = Q20 } #Wuwei
			1124 = { add_contested_owner  = Q20 } #Guyuan
			1125 = { add_contested_owner  = Q20 } #Yulin

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = Q20 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = Q20
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = Q20
				}
			}

			#Set custom focus tree to combat resistance
			Q20 = { load_focus_tree = pfp_formable_puppet_focus }
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			283 = { add_core_of = Q20 } #Gansu
			622 = { add_core_of = Q20 } #Shaanbei
			744 = { add_core_of = Q20 } #Xian
			756 = { add_core_of = Q20 } #Jiuquan
			1111 = { add_core_of = Q20 } #Yan'an
			1123 = { add_core_of = Q20 } #Wuwei
			1124 = { add_core_of = Q20 } #Guyuan
			1125 = { add_core_of = Q20 } #Yulin

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = Q20 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = Q20
					transfer_state_to = Q20
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = Q20
					set_state_controller_to = Q20
				}
			}

			Q20 = {
				remove_ideas = {
					pfp_puppet_resistance_idea_1
				}
			}
		}
	}
}
