#Scripted effect to release Northeastern U.S. (P20)
pfp_release_northeastern_us = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_enabled }

			#Add contestion to formable puppet
			357 = { add_contested_owner = P20 } #New England
			358 = { add_contested_owner = P20 } #New York
			359 = { add_contested_owner = P20 } #New Jersey
			360 = { add_contested_owner = P20 } #Pennsylvania
			361 = { add_contested_owner = P20 } #Maryland
			921 = { add_contested_owner = P20 } #New Hampshire
			990 = { add_contested_owner = P20 } #Vermont
			991 = { add_contested_owner = P20 } #Massachusetts
			993 = { add_contested_owner = P20 } #Connecticut
			992 = { add_contested_owner = P20 } #Rhode Island
			1023 = { add_contested_owner = P20 } #Delaware

			#Add claims to formable puppet
			357 = { add_claim_by = P20 } #New England
			358 = { add_claim_by = P20 } #New York
			359 = { add_claim_by = P20 } #New Jersey
			360 = { add_claim_by = P20 } #Pennsylvania
			361 = { add_claim_by = P20 } #Maryland
			921 = { add_claim_by = P20 } #New Hampshire
			990 = { add_claim_by = P20 } #Vermont
			991 = { add_claim_by = P20 } #Massachusetts
			993 = { add_claim_by = P20 } #Connecticut
			992 = { add_claim_by = P20 } #Rhode Island
			1023 = { add_claim_by = P20 } #Delaware

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = P20 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = P20
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = P20
				}
			}

			#Set custom focus tree to combat resistance
			P20 = { load_focus_tree = pfp_formable_puppet_focus }
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			357 = { add_core_of = P20 } #New England
			358 = { add_core_of = P20 } #New York
			359 = { add_core_of = P20 } #New Jersey
			360 = { add_core_of = P20 } #Pennsylvania
			361 = { add_core_of = P20 } #Maryland
			921 = { add_core_of = P20 } #New Hampshire
			990 = { add_core_of = P20 } #Vermont
			991 = { add_core_of = P20 } #Massachusetts
			993 = { add_core_of = P20 } #Connecticut
			992 = { add_core_of = P20 } #Rhode Island
			1023 = { add_core_of = P20 } #Delaware

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = P20 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = P20
					transfer_state_to = P20
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = P20
					set_state_controller_to = P20
				}
			}

			P20 = {
				remove_ideas = {
					pfp_puppet_resistance_idea_1
				}
			}
		}
	}
}
