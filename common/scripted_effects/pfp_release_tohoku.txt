#Scripted effect to release Tohoku (Q01)
pfp_release_tohoku = {
	hidden_effect = {
		#If resistance mechanics are enabled, formable puppets will contest and control its corresponding states.
		if = {
			limit = { has_global_flag = pfp_management_resistance_enabled }

			#Add contestion to formable puppet
			533 = { add_contested_owner = Q01 } #Kita-Tohoku
			1106 = { add_contested_owner = Q01 } #Minami Tohoku

			#Set formable puppet as controller of contested state
			every_state = {
				limit = { has_contested_owner = Q01 }
				
				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }
					transfer_state_to = Q01
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					set_state_controller_to  = Q01
				}
			}

			#Set custom focus tree to combat resistance
			Q01 = { load_focus_tree = pfp_formable_puppet_focus }
		}
		#If resistance mechanics are disabled, formable puppets will control and core its corresponding states.
		else = {
			#Add cores to formable puppet
			533 = { add_core_of = Q01 } #Kita-Tohoku
			1106 = { add_core_of = Q01 } #Minami Tohoku

			#Set formable puppet as owner and controller of core state
			every_state = {
				limit = { is_core_of = Q01 }

				#If ROOT owns the state, transfer control and ownership to formable puppet
				if = { 
					limit = { is_owned_by = ROOT }

					add_contested_owner = Q01
					transfer_state_to = Q01
				}

				#If ROOT does not own the state, transfer control to formable puppet
				else = {
					add_contested_owner = Q01
					set_state_controller_to = Q01
				}
			}

			Q01 = {
				remove_ideas = {
					pfp_puppet_resistance_idea_1
				}
			}
		}
	}
}
